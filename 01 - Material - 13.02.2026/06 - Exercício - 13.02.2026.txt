#include <iostream>

using namespace std;

int main(){

    int tempano[365];
    int menor = 40, maior = 0, media = 0, infmedia = 0;

    for(int i = 0; i < 365; i++){
        tempano[i] = rand() % (40 - 0 + 1) - 0;
        media += tempano[i];
    }

    media /= 365;

    for(int i = 0; i < 365; i++){
        if(tempano[i] > maior){
            maior = tempano[i];
        }
        if(tempano[i] < menor){
            menor = tempano[i];
        }
        if(tempano[i] < media){
            infmedia += 1;
        }
    }

    cout << "Registro de temperatura em Mogi Mirim no ano de 2025" << endl;
    for(int i = 0; i < 365; i++){
        cout << "Dia " << i + 1 << ": " << tempano[i] << endl;
    }

    cout << endl;
    cout << "Menor temperatura do ano: " << menor << endl;
    cout << "Maior temperatura do ano: " << maior << endl;
    cout << "Temperatura media anual: " << media << endl;
    cout << "Qtde de dias inferior a media: " << infmedia << endl;
}
