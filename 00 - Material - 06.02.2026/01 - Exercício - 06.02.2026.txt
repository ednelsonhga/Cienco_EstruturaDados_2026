#include <iostream>
#include <locale>

using namespace std;

int main()
{
    setlocale(LC_ALL, "portuguese");
    int peso;
    float altura, imc;

    cout << "Entre com seu peso em KG: ";
    cin >> peso;

    cout << "Entre com sua altura em CM: ";
    cin >> altura;

    altura /= 100;
    imc = peso/(altura*altura);

    if (imc < 20){
        cout << "Você está abaixo do peso!";
    }else if (imc < 25){
        cout << "Você está no peso ideal!";
    }else{
        cout << "Você está acima do peso!";
    }
}
