#include <iostream>
#include <locale>

using namespace std;

typedef struct animal {
    string nome;
    string especie;
    int idade;
    float peso;
};

int main() {

    setlocale (LC_ALL,"Portuguese");
    animal listanimais[10];

    cout << "Cadastro de Animais" << endl;
    cout << endl;

    for(int i = 0; i < 10; i++){

        cout << "Entre com o nome: ";
        getline(cin, listanimais[i].nome);

        cout << "Entre com a espécie: ";
        getline(cin, listanimais[i].especie);

        cout << "Entre com a idade: ";
        cin >> listanimais[i].idade;

        cout << "Entre com o peso: ";
        cin >> listanimais[i].peso;
        cout << endl;

        cin.ignore();

    }

    cout << endl;
    cout << "Segue dados preenchidos: " << endl;
    cout << endl;

    for(int i = 0; i < 10; i++){
        cout << "Cadastro 0" << i+1 << endl;
        cout << "Nome: " << listanimais[i].nome << endl;
        cout << "Espécie: " << listanimais[i].especie << endl;
        cout << "Idade: " << listanimais[i].idade << endl;
        cout << "Peso: " << listanimais[i].peso << endl;
        cout << endl;
    }
}
