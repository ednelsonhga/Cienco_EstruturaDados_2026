3)

#include <iostream>
#include <locale>
#include <cstdio>

using namespace std;

int main()
{

    setlocale(LC_ALL,"Portuguese");
    int n = 1;

    cout << "----- Análise de Sequência Numérica -----" << endl;
    cout << endl;
    cout << "-> Entre com a qtde de valores (entre 1 e 10): ";
    cin >> n;

    if(n < 1 || n > 10){
        do{
            cout << "Valor inválido! Entre com um valor entre 1 e 10: ";
            cin >> n;
        }while(n < 1 || n > 10);
    }
    cout << endl;

    int valores[n];
    bool par[n], impar[n], positivo[n], negativo[n], zero[n], mult3[n], mult5[n];
    int t_par = 0, t_impar = 0, t_positivo = 0, t_negativo = 0, t_zero = 0, t_mult3 = 0, t_mult5 = 0;
    int maior = 0, menor = 0;
    float media_par = 0, media_impar = 0;

    cout << "Entre com um valor inteiro:" << endl;
    for(int i = 0; i < n; i++){
        cout << "-> " << i + 1 << ": ";
        cin >> valores[i];
    }
    cout << endl;

    for(int i = 0; i < n; i++){
        if(valores[i] % 2 == 0){
            par[i] = 1;
            impar[i] = 0;
            t_par += 1;
            media_par += valores[i];
        }else{
            par[i] = 0;
            impar[i] = 1;
            t_impar += 1;
            media_impar += valores[i];
        }

        if(valores[i] > 0){
            positivo[i] = 1;
            negativo[i] = 0;
            zero[i] = 0;
            t_positivo += 1;
        }else if(valores[i] < 0){
            positivo[i] = 0;
            negativo[i] = 1;
            zero[i] = 0;
            t_negativo += 1;
        }else{
            positivo[i] = 0;
            negativo[i] = 0;
            zero[i] = 1;
            t_zero += 1;
        }

        if(valores[i] % 3 == 0){
            mult3[i] = 1;
            t_mult3 += 1;
        }else(mult3[i] = 0);

        if(valores[i] % 5 == 0){
            mult5[i] = 1;
            t_mult5 += 1;
        }else(mult5[i] = 0);

        if(valores[i] > maior){
            maior = valores[i];
        }
    }
    media_par /= t_par;
    media_impar /= t_impar;

    menor = maior;
    for(int i = 0; i < n; i++){
        if(valores[i] < menor){
            menor = valores[i];
        }
    }

    cout << "--------------------------------------------- RELATÓRIO ---------------------------------------------" << endl;
    cout << endl;
    printf("%-10s | %-10s | %-10s | %-10s | %-10s | %-10s | %-10s | %-10s\n", "Valores", "Par", "Ímpar", "Positivo", "Negativo", "Zero", "Múlt. de 3", "Múlt. de 5");
    printf("------------------------------------------------------------------------------------------------------\n");
    for(int i = 0; i < n; i++){
        printf("%-10d | %-10d | %-10d | %-10d | %-10d | %-10d | %-10d | %-10d\n", valores[i], par[i], impar[i], positivo[i], negativo[i], zero[i], mult3[i], mult5[i]);
    }
    printf("------------------------------------------------------------------------------------------------------\n");
    printf("%-10s | %-10d | %-10d | %-10d | %-10d | %-10d | %-10d | %-10d\n", "Totais", t_par, t_impar, t_positivo, t_negativo, t_zero, t_mult3, t_mult5);
    cout << endl;
    cout << "Maior: " << maior << endl;
    cout << "Menor: " << menor << endl;
    cout << "Média dos pares: " << media_par << endl;
    cout << "Média dos ímpares: " << media_impar << endl;
    cout << endl;
    cout << "------------------------------------------------------------------------------------------------------" << endl;
}
