#include <iostream>
#include <locale>

using namespace std;

int main()
{

    setlocale(LC_ALL,"Portuguese");
    int a = 0, b = 0, c = 0, maior = 0;

    cout << "----- CLASSIFICAÇÃO DE TRIÂNGULOS -----" << endl;
    cout << endl;

    cout << "Entre com 3 valores inteiro:" << endl;
    cout << "A: ";
    cin >> a;
    cout << "B: ";
    cin >> b;
    cout << "C: ";
    cin >> c;

    if(a < b + c && b < a + c && c < a + b){

        cout << endl;
        cout << "Resultado:" << endl;
        cout << " > É um triângulo." << endl;

        if (a == b && b == c){
            cout << " > É equilátero." << endl;
        }else if(a != b && b!= c){
            cout << " > É escaleno." << endl;
        }else{
            cout << " > É isósceles." << endl;
        }

        if(a > maior){
            maior = a;
        }else if(b > maior){
            maior = b;
        }else if(c > maior){
            maior = c;
        }

        if(maior * maior == a * a + b * b){
            cout << " > É retângulo." << endl;
        }else if(maior * maior > a * a + b * b){
            cout << " > É obtusângulo." << endl;
        }else{
            cout << " > É acutângulo." << endl;
        }

    }else{
        cout << "Não forma um triângulo." << endl;
    }
}

{

            t_impar += 1;
            media_impar += valores[i];

        if(valores[i] > 0){
            positivo[i] = 1;
            negativo[i] = 0;
            zero[i] = 0;
            t_positivo += 1;

            positivo[i] = 0;
            negativo[i] = 1;

            positivo[i] = 0;

            zero[i] = 1;
            t_zero += 1;

            mult3[i] = 1;

        }

    media_impar /= t_impar;

#include <iostream>
#include <locale>

using namespace std;

int main()
{
    setlocale(LC_ALL,"Portuguese");
    int n = 0;

    cout << "----- ANÁLISE DE VETOR -----" << endl;
    cout << endl;
    cout << "-> Tamanho do vetor (entre 10 e 50): ";
    cin >> n;

    if(n < 10 || n > 50){
        do{
            cout << "Valor inválido. Entre com um valor entre 10 e 50: ";
            cin >> n;
        }while(n < 10 || n > 50);
    }
    cout << endl;

    int vetor[n], vetor_reorganizado[n], repeticoes[n], qtdepar = 0, qtdeimpar = 0, qtdenegaivo = 0, qtdepositivo = 0, qtdezero = 0, qtdemult2 = 0, qtdemult3 = 0, qtdemult5 = 0, maior = -99999, menor = 0;
    float media = 0;

    cout << "Preenchimento do vetor:" << endl;
    for(int i = 0; i < n; i++){
        cout << "-> " << i + 1 << ": ";
        cin >> vetor[i];
    }
    cout << endl;

    for(int i = 0; i < n; i++){
        if(vetor[i] % 2 == 0){
            vetor_reorganizado[qtdepar] = vetor[i];
            qtdepar += 1;
            qtdemult2 += 1;
        }else{
            qtdeimpar += 1;
        }

        if(vetor[i] > 0){
            qtdepositivo += 1;
        }else if(vetor[i] < 0){
            qtdenegaivo += 1;
        }else{
            qtdezero += 1;
        }

        if(vetor[i] % 3 == 0){
            qtdemult3 += 1;
        }

        if(vetor[i] % 5 == 0){
            qtdemult5 += 1;
        }

        if(vetor[i] > maior){
            maior = vetor[i];
        }

        media += vetor[i];
    }
    media /= n;

    menor = maior;
    for(int i = 0; i < n; i++){
        if(vetor[i] < menor){
            menor = vetor[i];
        }
    }

    int y = qtdepar;
    for(int i = n; i > 0; i--){
        if(vetor[i] % 2 > 0){
            vetor_reorganizado[y] = vetor[i];
            y += 1;
        }
    }

    for(int i = 0; i < n; i++){
        repeticoes[i] = 0;
    }

    for(int i = 0; i < n; i++){
        for(int y = 0; y < n; y++){
            if(vetor[i] == vetor[y]){
                repeticoes[i] += 1;
            }
        }
    }

    cout << "----- Resultados -----" << endl;
    cout << endl;
    cout << "Vetor:" << endl;

    for(int i = 0; i < n; i++){
        cout << "-> " << i + 1 << ": " << vetor[i] << endl;
    }

    cout << endl;
    cout << "Vetor reorganizado:" << endl;
    for(int i = 0; i < n; i++){
        cout << "-> " << i + 1 << ": " << vetor_reorganizado[i] << endl;
    }

    cout << endl;
    cout << "Totais:" << endl;
    cout << "-> Pares: " << qtdepar << endl;
    cout << "-> Ímpares: " << qtdeimpar << endl;
    cout << "-> Positivos: " << qtdepositivo << endl;
    cout << "-> Negativos: " << qtdenegaivo << endl;
    cout << "-> Zeros: " << qtdezero << endl;
    cout << "-> Múltiplos de 2: " << qtdemult2 << endl;
    cout << "-> Múltiplos de 3: " << qtdemult3 << endl;
    cout << "-> Múltiplos de 5: " << qtdemult5 << endl;
    cout << endl;
    cout << "Outras Análises:" << endl;
    cout << "-> Maior valor: " << maior << endl;
    cout << "-> Menor valor: " << menor << endl;
    cout << "-> Média: " << media << endl;
}

#include <iostream>
#include <locale>

using namespace std;

struct funcionario{
    int id;
    string nome;
    float salario;
    float novo_salario;
    int idade;
    int tempo_empresa;
};

int main()
{
    setlocale(LC_ALL,"Portuguese");
    int n = 0, func_sal_acima = 0, func_idade_50 = 0, func_10_empresa = 0;
    float media_salarial = 0, maior_salario = 0, menor_salario = 0, percentual = 0;

    cout << "----- GESTÃO DE FUNCIONÁRIOS -----" << endl;
    cout << endl;
    cout << "-> Número de funcionários (Entre 5 e 50): ";
    cin >> n;
    cout << endl;

    if(n < 5 || n > 50){
        do{
            cout << "Valor inválido! Entre com um valor entre 5 e 50: ";
            cin >> n;
        }while(n < 5 || n > 50);
    }

    funcionario v_funcionario[5];

    cout << "Preenchimento dos Dados dos Funcionários" << endl;
    cout << endl;

    for(int i = 0; i < n; i++){
        cout << "Funcionário - ID " << i + 1 << ":" << endl;
        v_funcionario[i].id = i + 1;

        cin.ignore();
        cout << "-> Nome: ";
        getline(cin, v_funcionario[i].nome);

        cout << "-> Salário: R$ ";
        cin >> v_funcionario[i].salario;
        media_salarial += v_funcionario[i].salario;

        cout << "-> Idade: ";
        cin >> v_funcionario[i].idade;

        cout << "-> Anos de Empresa: ";
        cin >> v_funcionario[i].tempo_empresa;
        cout << endl;
    }

    media_salarial /= n;

    for(int i = 0; i < n; i++){
        if(v_funcionario[i].salario > maior_salario){
            maior_salario = v_funcionario[i].salario;
        }
    }

    menor_salario = maior_salario;

    for(int i = 0; i < n; i++){
        if(v_funcionario[i].salario < menor_salario){
            menor_salario = v_funcionario[i].salario;
        }
    }

    for(int i = 0; i < n; i++){
        if(v_funcionario[i].salario > media_salarial){
            func_sal_acima += 1;
        }

        if(v_funcionario[i].idade > 50){
            func_idade_50 += 1;
        }

        if(v_funcionario[i].tempo_empresa > 10){
            func_10_empresa += 1;
        }
    }

    for(int i = 0; i < n; i++){
        percentual = 0;

        if(v_funcionario[i].tempo_empresa < 3){
            percentual += 0.05;
        }else if(v_funcionario[i].tempo_empresa <= 10){
            percentual += 0.10;
        }else{
            percentual += 0.20;
        }

        if(v_funcionario[i].idade > 50){
            percentual += 0.05;
        }

        if(v_funcionario[i].salario < media_salarial){
            percentual += 0.05;
        }

        v_funcionario[i].novo_salario = v_funcionario[i].salario + v_funcionario[i].salario * percentual;
    }

    cout << "Lista Preenchida" << endl;
    cout << endl;
    for(int i = 0; i < n; i++){
        cout << "Funcionário - ID: " << v_funcionario[i].id << endl;
        cout << "-> Nome: " << v_funcionario[i].nome << endl;
        cout << "-> Salário: R$ " << v_funcionario[i].salario << endl;
        cout << "-> Salário Reajustado: R$ " << v_funcionario[i].novo_salario << endl;
        cout << "-> Idade: " << v_funcionario[i].idade << endl;
        cout << "-> Anos de Empresa: " << v_funcionario[i].tempo_empresa << endl;
        cout << endl;
    }
}
